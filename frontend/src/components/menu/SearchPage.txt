// SearchPage.js que saqué
import React from 'react';

const SearchPage = ({ searchResults }) => {
  // Verificar si searchResults es undefined o está vacío
  if (!searchResults || searchResults.length === 0) {
    // Si searchResults es undefined o está vacío, muestra un mensaje o cualquier contenido alternativo
    return <p>No hay resultados de búsqueda.</p>;
  }

  return (
    <div>
      <h1>Resultados de búsqueda: hola</h1>
      <div className="content">
        <div className="row">
          {searchResults.map((product) => (
            <div key={product.id} className="col-md-4 mb-4">
              <div className="card">
                <img src={product.imagen} className="card-img-top" alt={product.alt} />
                <div className="card-body">
                  
                  <h5 className="card-title">{product.nombre}</h5>

                  <p className="card-text">
                    Precio: {product.precio}
                  </p>
                </div>
              </div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};

export default SearchPage;
